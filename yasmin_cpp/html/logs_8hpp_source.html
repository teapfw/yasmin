56inline const char *extract_filename(const char *path) {
57 const char *filename = std::strrchr(path, '/');
58 if (!filename) {
59 filename = std::strrchr(path, '\\'); // handle Windows-style paths
60 }
61 return filename ? filename + 1 : path;
